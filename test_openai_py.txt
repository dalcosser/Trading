from openai import OpenAI

client = OpenAI()  # automatically reads your OPENAI_API_KEY

response = client.chat.completions.create(
    model="gpt-4-turbo",
    messages=[
        {"role": "system", "content": "You are a helpful coding assistant."},
        {"role": "user", "content": "Write a Python function that calculates a 5-minute VWAP from OHLCV data in pandas."}
    ]
)

print(response.choices[0].message.content)
